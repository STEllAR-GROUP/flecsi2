350  iterator insert(iterator i, size_type n, const T & t) const {
351  const auto b = brki();
352  auto & a = o->add;
353  if(i < b) {
354  const size_type
355  // used elements assigned from t:
356  filla = std::min(n, b.index() - i.index()),
357  fillx = n - filla, // other spaces to fill
358  fillc = std::min(fillx, o->del), // slack spaces constructed from t
359  fillo = fillx - fillc, // overflow copies of t
360  // slack spaces move-constructed from used elements:
361  mvc = std::min(o->del - fillc, filla);
362 
363  // Perform the moves and fills mostly in reverse order.
364  T *const s0 = s.data(), *const ip = s0 + i.index(),
365  *const bp = s0 + b.index();
366  // FIXME: Is there a way to do just one shift?
367  a.insert(a.begin(), fillo, t); // first to reduce the volume of shifting
368  a.insert(a.begin() + fillo, bp - (filla - mvc), bp);
369  std::uninitialized_move_n(
370  bp - filla, mvc, std::uninitialized_fill_n(bp, fillc, t));
371  o->del -= fillc + mvc; // before copies that might throw
372  std::fill(ip, std::move_backward(ip, bp - filla, bp), t);
373  }
374  else {
375  if(i == b)

567  void commit() const {
568  // To move each element before overwriting it, we propagate moves outward
569  // from each place where the movement switches from rightward to leftward.
570  const auto all = acc.get_base().span();
571  const std::size_t n = size();
572  // Read and write cursors. It would be possible, if ugly, to run cursors
573  // backwards for the rightward-moving portions and do without the stack.
574  std::size_t is = 0, js = 0, id = 0, jd = 0;
575  raw_row rs = raw_get(is), rd = raw_get(id);
576  struct work {
577  void operator()() const {
578  if(assign)
579  *w = std::move(*r);
580  else
581  new(w) T(std::move(*r));
582  }
583  T *r, *w;
584  bool assign;
585  };
586  std::stack<work> stk;
587  const auto back = [&stk] {
588  for(; !stk.empty(); stk.pop())
589  stk.top()();
590  };
591  for(;; ++js, ++jd) {
592  while(js == rs.size()) { // js indexes the split row
593  if(++is == n)
594  break;
595  rs = raw_get(is);
596  js = 0;
597  }
598  if(is == n)
599  break;
600  while(jd == rd.s.size()) { // jd indexes the span (including slack space)
601  if(++id == n)
602  break; // we can write off the end of the last
603  else
604  rd = raw_get(id);
605  jd = 0;
606  }
607  const auto r = &rs[js], w = rd.s.data() + jd;
608  flog_assert(w != all.end(),
609  "ragged entries for last " << n - is << " rows overrun allocation for "
610  << all.size() << " entries");
611  if(r != w)
612  stk.push({r, w, jd < rd.brk()});
613  // Perform this move, and any already queued, if it's not to the right.
614  // Offset real elements by one position to come after insertions.
615  if(rs.s.data() + std::min(js + 1, rs.brk()) > w)
616  back();
617  }
618  back();
619  if(jd < rd.brk())
620  rd.destroy(jd);
621  while(++id < n)
622  raw_get(id).destroy();
623  auto & off = acc.get_offsets();
624  std::size_t delta = 0; // may be "negative"; aliasing is implausible
625  for(is = 0; is < n; ++is) {
626  auto & ov = (*over)[is];
627  off(is) += delta += ov.add.size() - ov.del;
628  ov.add.clear();
629  }
630  sz = data::partition::make_row(
631  run::context::instance().color(), grow(acc.total(), all.size()));
632  }
