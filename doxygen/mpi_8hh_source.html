177 template<typename F>
178 inline auto
179 one_to_allv(F const & f, MPI_Comm comm = MPI_COMM_WORLD) {
180  using return_type = decltype(f(int(0), int(1)));
181 
182  auto [rank, size] = info(comm);
183 
184  std::vector<std::vector<std::byte>> data(size);
185 
186  if(rank == 0) {
187  std::vector<MPI_Request> requests;
188  requests.reserve(2 * (size - 1));
189 
190  for(size_t r{1}; r < std::size_t(size); ++r) {
191  data[r] = serial_put(f(r, size));
192  const std::size_t bytes = data[r].size();
193 
194  requests.resize(requests.size() + 1);
195  MPI_Isend(&bytes, 1, type<std::size_t>(), r, 0, comm, &requests.back());
196  requests.resize(requests.size() + 1);
197  MPI_Isend(
198  data[r].data(), bytes, type<std::byte>(), r, 0, comm, &requests.back());
199  } // for
200 
201  std::vector<MPI_Status> status(requests.size());
202  MPI_Waitall(requests.size(), requests.data(), status.data());
203  }
204  else {
205  std::size_t bytes{0};
206  MPI_Status status;
207  MPI_Recv(&bytes, 1, type<std::size_t>(), 0, 0, comm, &status);
208  std::vector<std::byte> data(bytes);
209  MPI_Recv(data.data(), bytes, type<std::byte>(), 0, 0, comm, &status);
210  auto const * p = data.data();
211  return serial_get<return_type>(p);
212  } // if
213 
214  return f(0, size);
215 } // one_to_allv

233 template<typename F>
234 inline auto
235 all_to_allv(F const & f, MPI_Comm comm = MPI_COMM_WORLD) {
236  using return_type = decltype(f(int(0), int(1)));
237 
238  auto [rank, size] = info(comm);
239 
240  std::vector<std::size_t> counts;
241  counts.reserve(size);
242  std::vector<std::size_t> bytes(size);
243 
244  for(std::size_t r{0}; r < std::size_t(size); ++r) {
245  counts.emplace_back(serial_size<return_type>(f(r, size)));
246  } // for
247 
248  MPI_Alltoall(counts.data(),
249  1,
250  type<std::size_t>(),
251  bytes.data(),
252  1,
253  type<std::size_t>(),
254  comm);
255 
256  std::vector<std::vector<std::byte>> recv_bufs(size);
257 
258  {
259  std::vector<MPI_Request> requests;
260  requests.reserve(2 * size);
261  std::vector<std::vector<std::byte>> send_bufs(size);
262 
263  for(std::size_t r{0}; r < std::size_t(size); ++r) {
264  if(bytes[r] > 0) {
265  recv_bufs[r].resize(bytes[r]);
266  requests.resize(requests.size() + 1);
267  MPI_Irecv(recv_bufs[r].data(),
268  recv_bufs[r].size(),
269  type<std::byte>(),
270  r,
271  0,
272  comm,
273  &requests.back());
274  } // if
275  } // for
276 
277  bytes.clear();
278 
279  for(std::size_t r{0}; r < std::size_t(size); ++r) {
280  if(counts[r] > 0) {
281  requests.resize(requests.size() + 1);
282  send_bufs[r] = serial_put(f(r, size));
283  MPI_Isend(send_bufs[r].data(),
284  send_bufs[r].size(),
285  type<std::byte>(),
286  r,
287  0,
288  comm,
289  &requests.back());
290  } // if
291  } // for
292 
293  counts.clear();
294 
295  std::vector<MPI_Status> status(requests.size());
296  MPI_Waitall(requests.size(), requests.data(), status.data());
297  } // scope
298 
299  std::vector<return_type> result;
300  result.reserve(size);
301  for(std::size_t r{0}; r < std::size_t(size); ++r) {
302  if(recv_bufs[r].size() > 0) {
303  auto const * p = recv_bufs[r].data();
304  result.emplace_back(serial_get<return_type>(p));
305  }
306  } // for
307 
308  return result;
309 } // all_to_allv
