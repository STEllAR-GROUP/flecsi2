45 template<typename Definition, size_t from_dim, size_t to_dim, size_t thru_dim>
46 std::set<size_t>
47 entity_neighbors(const Definition & md, size_t entity_id) {
48  // Get the vertices of the requested id
49  auto vertices = md.entities_set(from_dim, 0, entity_id);
50 
51  // Put the results into set form
52  std::set<size_t> neighbors;
53 
54  // Go through the entities of the to_dim
55  for(size_t e(0); e < md.num_entities(to_dim); ++e) {
56 
57  // Skip the input id if the dimensions are the same
58  if(from_dim == to_dim && e == entity_id) {
59  continue;
60  } // if
61 
62  // Get the vertices that define the current entity from the to_dim
63  auto other = md.entities_set(to_dim, 0, e);
64 
65  // Get the intersection set
66  auto intersect = util::set_intersection(vertices, other);
67 
68  // Add this entity id if the intersection shares at least
69  // intersections vertices
70  if(intersect.size() > thru_dim) {
71  neighbors.insert(e);
72  } // if
73  } // for
74 
75  return neighbors;
76 } // entity_neighbors

94 template<typename Definition,
95  size_t from_dim,
96  size_t to_dim,
97  size_t thru_dim,
98  typename U,
99  typename = std::enable_if_t<util::is_iterative_container_v<U>>>
100 std::set<size_t>
101 entity_neighbors(const Definition & md, U && indices) {
102  flog_assert(from_dim == to_dim, "from_dim does not equal to to_dim");
103 
104  // Closure should include the initial set
105  std::set<size_t> closure(
106  std::forward<U>(indices).begin(), std::forward<U>(indices).end());
107 
108  using entityid = size_t;
109  using vertexid = size_t;
110 
111  // essentially vertex to cell and cell to cell through vertices
112  // connectivity.
113  std::map<vertexid, std::vector<entityid>> vertex2entities;
114  std::map<entityid, std::vector<entityid>> entity_neighbors;
115 
116  // build global entity to entity through # of shared vertices connectivity
117  for(size_t cell(0); cell < md.num_entities(from_dim); ++cell) {
118  std::map<entityid, size_t> cell_counts;
119  for(auto vertex : md.entities(from_dim, 0, cell)) {
120  // build vertex to cell connectivity, O(n_cells * n_polygon_sides)
121  // of insert().
122  vertex2entities[vertex].push_back(cell);
123 
124  // Count the number of times this cell shares a common vertex with
125  // some other cell. O(n_cells * n_polygon_sides * vertex to cells degrees)
126  for(auto other : vertex2entities[vertex]) {
127  // for (auto other : md.entities(0, from_dim, vertex)) {
128  if(other != cell)
129  cell_counts[other] += 1;
130  }
131  }
132 
133  for(auto count : cell_counts) {
134  if(count.second > thru_dim) {
135  // append cell to cell through "dimension" connectivity, we need to
136  // add both directions
137  entity_neighbors[cell].push_back(count.first);
138  entity_neighbors[count.first].push_back(cell);
139  }
140  }
141  }
142 
143  for(auto i : indices) {
144  for(auto n : entity_neighbors[i]) {
145  closure.insert(n);
146  }
147  }
148 
149  return closure;
150 } // entity_neighbors
