245 struct task_wrapper {
246 
247  using Traits = util::function_traits<decltype(F)>;

257  static RETURN execute(const Legion::Task * task,
258  const std::vector<Legion::PhysicalRegion> & regions,
259  Legion::Context context,
260  Legion::Runtime * runtime) {
261  {
262  log::devel_guard guard(task_wrapper_tag);
263  flog_devel(info) << "In execute_user_task" << std::endl;
264  }
265 
266  // Unpack task arguments
267  auto task_args = detail::tuple_get<param_tuple>(*task);
268 
269  namespace ann = util::annotation;
270  auto tname = util::symbol<F>();
271  unbind_accessors ub(task_args);
272  (ann::rguard<ann::execute_task_bind>(tname)),
273  bind_accessors(runtime, context, regions, task->futures)(task_args);
274 
275  if constexpr(std::is_same_v<RETURN, void>) {
276  (ann::rguard<ann::execute_task_user>(tname)),
277  apply(F, std::forward<param_tuple>(task_args));
278  ann::rguard<ann::execute_task_unbind> ann_guard(tname);
279  ub();
280  }
281  else {
282  RETURN result = (ann::rguard<ann::execute_task_user>(tname),
283  apply(F, std::forward<param_tuple>(task_args)));
284  ann::rguard<ann::execute_task_unbind> ann_guard(tname);
285  ub();
286  return result;
287  } // if
