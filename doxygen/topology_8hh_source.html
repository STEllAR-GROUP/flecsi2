117  // Return whether a copy is needed.
118  template<std::size_t P>
119  bool ghost(field_id_t i) {
120  constexpr auto n = privilege_count(P);
121  static_assert(n > 1, "need shared/ghost privileges");
122  constexpr auto g = get_privilege(n - 1, P);
123  constexpr bool gr = privilege_read(g),
124  sw = privilege_write(get_privilege(n - 2, P));
125  // The logic here is constructed to allow a single read/write set access:
126  // writing to ghosts, or reading from them without also writing to shared,
127  // clears the dirty bit, and otherwise writing to shared sets it.
128  // Otherwise, it retains its value (and we don't copy).
129  return (privilege_write(g) || !sw && gr ? dirty.erase(i)
130  : sw && !dirty.insert(i).second) &&
131  gr;
132  }
