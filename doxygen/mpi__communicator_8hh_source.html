92  // the indices of the actual request. Each rank that receives
93  // the request will try to provide information about the
94  // non size_t max values in the request. The others will
95  // be ignored.

154 
155  // For now, we need two arrays for each all-to-all communication:
156  // One for rank ownership of the request indices, and one
157  // for the offsets. We could probably combine these. However,
158  // we would probably have to define a custom MPI type. It

160  // The input offsets do not need to be initialized because

163  std::vector<size_t> info_offsets(colors * max_request_indices);

166  std::fill(input_indices.begin(),
167  input_indices.end(),
168  std::numeric_limits<size_t>::max());
169

171  // on indices that are shared with other processes.
172  std::vector<std::set<size_t>> local(primary.size());
173 
174  // See if we can fill any requests...
175  for(size_t r(0); r < colors; ++r) {
176 
177  // Ignore our rank
178  if(r == color) {
179  continue;
180  } // if
181 
182  // These array slices are just for convenience.
183  size_t * info = &info_indices[r * max_request_indices];
184  size_t * offset = &input_offsets[r * max_request_indices];
185  size_t * input = &input_indices[r * max_request_indices];
186 
187  // See which requests we can fulfill.
188  for(size_t i(0); i < max_request_indices; ++i) {
189 
190  auto match = primary.find(info[i]);
191 
192  if(match != primary.end()) {
193  // This is a match, i.e., we own this entity, so we can
194  // set the rank (ownership) and offset.
195  input[i] = color;
196  offset[i] = std::distance(primary.begin(), match);
197 
198  // We also need to register that this index is shared
199  // with other ranks
200  local[offset[i]].insert(r);
201  } // if
202  } // for
203  } // for
204 
205  // Send the indices information back to all ranks.
206  MPI_Alltoall(&input_indices[0],
207  max_request_indices,
208  util::mpi::type<std::size_t>(),
209  &info_indices[0],
210  max_request_indices,
211  util::mpi::type<std::size_t>(),
212  comm);
213 
214  // Send the offsets information back to all ranks.
215  MPI_Alltoall(&input_offsets[0],
216  max_request_indices,
217  util::mpi::type<std::size_t>(),
218  &info_offsets[0],
219  max_request_indices,
220  util::mpi::type<std::size_t>(),
221  comm);
222

224 
225  // Collect all of the information for the remote entities.
226  for(size_t r(0); r < colors; ++r) {
227  // Skip these (we already know them!)
228  if(r == color) {
229  continue;
230  } // if
231 
232  // Another slice for convenience.
233  size_t * ranks = &info_indices[r * max_request_indices];
234  size_t * offsets = &info_offsets[r * max_request_indices];
235 
236  for(size_t i(0); i < max_request_indices; ++i) {
237 
238  if(ranks[i] != std::numeric_limits<size_t>::max()) {
239  // If this is not size_t max, this rank answered our request
240  // and we can set the information.
241  remote.insert(
242  entity_info(request_indices_vector[i], ranks[i], offsets[i]));
243  } // if
244  } // for
245  } // for
246 
247  return std::make_pair(local, remote);
248  } // get_primary_info

389  std::vector<std::set<size_t>> get_entity_info(
390  const std::set<entity_info> & entities,
391  const std::vector<std::set<size_t>> & request_indices) {
392  auto colors = size();
393 
394  // Collect the size of each rank request to send.
395  std::vector<size_t> send_cnts(colors, 0);
396  for(size_t r(0); r < colors; ++r) {
397  send_cnts[r] = request_indices[r].size();
398  } // for
399 
400  // Send the request size (in indices) to each rank.
401  std::vector<size_t> recv_cnts(colors);
402  MPI_Alltoall(&send_cnts[0],
403  1,
404  util::mpi::type<std::size_t>(),
405  &recv_cnts[0],
406  1,
407  util::mpi::type<std::size_t>(),
408  comm);
409 
410  // Start receive operations (non-blocking).
411  std::vector<std::vector<size_t>> rbuffers(colors);
412  std::vector<MPI_Request> requests;
413  for(size_t r(0); r < colors; ++r) {
414  if(recv_cnts[r]) {
415  rbuffers[r].resize(recv_cnts[r]);
416  requests.push_back({});
417  MPI_Irecv(&rbuffers[r][0],
418  recv_cnts[r],
419  util::mpi::type<std::size_t>(),
420  r,
421  0,
422  comm,
423  &requests[requests.size() - 1]);
424  } // if
425  } // for
426 
427  // Start send operations (blocking is ok here).
428  std::vector<std::vector<size_t>> sbuffers(colors);
429  for(size_t r(0); r < colors; ++r) {
430  if(send_cnts[r]) {
431  std::copy(request_indices[r].begin(),
432  request_indices[r].end(),
433  std::back_inserter(sbuffers[r]));
434 
435  MPI_Send(&sbuffers[r][0],
436  send_cnts[r],
437  util::mpi::type<std::size_t>(),
438  r,
439  0,
440  comm);
441  } // if
442  } // for
443 
444  // Create a map version of the entity info for lookups below.
445  std::unordered_map<size_t, entity_info> entity_info_map;
446  for(auto i : entities) {
447  entity_info_map[i.id] = i;
448  } // for
449 
450  // Wait on the receive operations
451  std::vector<MPI_Status> status(requests.size());
452  MPI_Waitall(requests.size(), &requests[0], &status[0]);
453 
454  // Set the offsets for each requested index in the send buffer.
455  for(size_t r(0); r < colors; ++r) {
456  sbuffers[r].resize(rbuffers[r].size());
457 
458  size_t offset(0);
459  for(auto i : rbuffers[r]) {
460  sbuffers[r][offset++] = entity_info_map[i].offset;
461  } // for
462  } // for
463 
464  // we are reusing requests in the following code, need to clear the content
465  // first.
466  requests.clear();
467 
468  // Start receive operations (non-blocking) to get back the
469  // offsets we requested.
470  for(size_t r(0); r < colors; ++r) {
471  // If we sent a request, prepare to receive an answer.
472  if(send_cnts[r]) {
473  // We're done with our receive buffers, so we can re-use them.
474  rbuffers[r].resize(send_cnts[r], 0);
475  requests.push_back({});
476  MPI_Irecv(&rbuffers[r][0],
477  send_cnts[r],
478  util::mpi::type<std::size_t>(),
479  r,
480  0,
481  comm,
482  &requests[requests.size() - 1]);
483  } // if
484  } // for
485 
486  // Start send operations (blocking is probably ok here).
487  for(size_t r(0); r < colors; ++r) {
488  // If we received a request, prepare to send an answer.
489  if(recv_cnts[r]) {
490  MPI_Send(&sbuffers[r][0],
491  recv_cnts[r],
492  util::mpi::type<std::size_t>(),
493  r,
494  0,
495  comm);
496  } // if
497  } // for
498 
499  // Wait on the receive operations
500  status.resize(requests.size());
501  MPI_Waitall(requests.size(), &requests[0], &status[0]);
502 
503  std::vector<std::set<size_t>> remote(colors);
504  for(size_t r(0); r < colors; ++r) {
505  for(size_t i(0); i < send_cnts[r]; ++i) {
506  remote[r].insert(rbuffers[r][i]);
507  } // for
508  } // for
509 
510  return remote;
511  } // get_entity_info
