226  auto print_usage = [this](std::string program,
227  auto const & master,
228  auto const & flecsi) {
229  if(process_ == 0) {
230  std::cout << "Usage: " << program << " ";
231 
232  size_t positional_count = positional_desc_.max_total_count();
233  size_t max_label_chars = std::numeric_limits<size_t>::min();
234 
235  for(size_t i{0}; i < positional_count; ++i) {
236  std::cout << "<" << positional_desc_.name_for_position(i) << "> ";
237 
238  const size_t size = positional_desc_.name_for_position(i).size();
239  max_label_chars = size > max_label_chars ? size : max_label_chars;
240  } // for
241 
242  max_label_chars += 2;
243 
244  std::cout << std::endl << std::endl;
245 
246  if(positional_count) {
247  std::cout << "Positional Options:" << std::endl;
248 
249  for(size_t i{0}; i < positional_desc_.max_total_count(); ++i) {
250  auto const & name = positional_desc_.name_for_position(i);
251  auto help = positional_help_.at(name);
252  std::cout << " " << name << " ";
253  std::cout << std::string(max_label_chars - name.size() - 2, ' ');
254 
255  if(help.size() > 78 - max_label_chars) {
256  std::string first = help.substr(
257  0, help.substr(0, 78 - max_label_chars).find_last_of(' '));
258  std::cout << first << std::endl;
259  help =
260  help.substr(first.size() + 1, help.size() - first.size() + 1);
261 
262  while(help.size() > 78 - max_label_chars) {
263  std::string part = help.substr(
264  0, help.substr(0, 78 - max_label_chars).find_last_of(' '));
265  std::cout << std::string(max_label_chars + 1, ' ') << part
266  << std::endl;
267  help = help.substr(part.size() + 1, help.size() - part.size());
268  } // while
269 
270  std::cout << std::string(max_label_chars + 1, ' ') << help
271  << std::endl;
272  }
273  else {
274  std::cout << help << std::endl;
275  } // if
276 
277  } // for
278 
279  std::cout << std::endl;
280  } // if
281 
282  std::cout << master << std::endl;
283  std::cout << flecsi << std::endl;
284 
285 #if defined(FLECSI_ENABLE_FLOG)
286  auto const & tm = log::state::instance().tag_map();

289  std::cout << "Available FLOG Tags (FleCSI Logging Utility):"

302  boost::program_options::store(parsed, vm);

331  } // if
332  } // if
333  } // for

336  std::string error(e.what());
337 
338  auto pos = error.find("--");
339  if(pos != std::string::npos) {
340  error.replace(pos, 2, "");
341  } // if

343  std::cerr << FLOG_COLOR_LTRED << "ERROR: " << FLOG_COLOR_RED << error
344  << "!!!" << FLOG_COLOR_PLAIN << std::endl
345  << std::endl;
